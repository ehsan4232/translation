\section{مراحل کار CDN}
\begin{enumerate}
\setLTR
\item[۴.] CDN تصویر را کش کرده و به کاربر الف برمی‌گرداند. تصویر تا زمان انقضای TTL در CDN ذخیره می‌شود.
\item[۵.] کاربر ب درخواستی برای دریافت همان تصویر ارسال می‌کند.
\item[۶.] تصویر تا زمانی که TTL منقضی نشده باشد از کش برگردانده می‌شود.
\end{enumerate}

\section{ملاحظات استفاده از CDN}
\begin{itemize}
\item \textbf{هزینه}: CDNها توسط ارائه‌دهندگان شخص ثالث اداره می‌شوند و شما بابت انتقال داده به داخل و خارج CDN هزینه پرداخت می‌کنید. کش کردن فایل‌های کم‌استفاده مزیت چندانی ندارد و بهتر است آن‌ها را از CDN خارج کنید.

\item \textbf{تنظیم زمان انقضای مناسب کش}: برای محتوای حساس به زمان، تنظیم زمان انقضای کش اهمیت دارد. این زمان نباید خیلی طولانی یا خیلی کوتاه باشد. اگر خیلی طولانی باشد، محتوا ممکن است دیگر تازه نباشد. اگر خیلی کوتاه باشد، باعث بارگیری مکرر محتوا از سرورهای مبدأ به CDN می‌شود.

\item \textbf{راه‌حل جایگزین برای CDN}: باید در نظر بگیرید که وب‌سایت/برنامه شما چگونه با خرابی CDN کنار می‌آید. در صورت قطعی موقت CDN، کلاینت‌ها باید بتوانند مشکل را تشخیص داده و منابع را مستقیماً از مبدأ درخواست کنند.

\item \textbf{ابطال فایل‌ها}: می‌توانید فایلی را قبل از انقضای آن از CDN حذف کنید با یکی از روش‌های زیر:
\begin{itemize}
\item ابطال آبجکت CDN با استفاده از APIهای ارائه‌شده توسط فروشندگان CDN
\item استفاده از نسخه‌بندی آبجکت برای ارائه نسخه متفاوتی از آن. برای نسخه‌بندی می‌توانید پارامتری به URL اضافه کنید، مثلاً یک شماره نسخه: 
\begin{latin}
image.png?v=2
\end{latin}
\end{itemize}
\end{itemize}

\section{بررسی ترجمه}
\begin{itemize}
\item تمام مفاهیم به صورت روان و قابل فهم به فارسی ترجمه شده‌اند.
\item اصطلاحات فنی مانند CDN، TTL و API به صورت اصلی باقی مانده‌اند.
\item ساختار جمله‌بندی‌ها برای خوانایی بهینه شده است.
\item اعداد و فرمت‌بندی به درستی رعایت شده‌اند.
\item جهت نوشتار (RTL) به درستی اعمال شده است.
\end{itemize}
